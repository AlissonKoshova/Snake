using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Snakeke
{
    class Program
    {
        static void Main(string[] args)
        {
            Point p = new Point();
            p.x = 1;
            p.y = 2;
            p.s = '%';
            Draw(p.x, p.y, p.s);
            Draw(1, 3, '*');
            Draw(4, 5, '%');
            Console.ReadLine();
        }
        static void Draw (int x,int y, char s)
        {
            Console.SetCursorPosition(x, y);
            Console.Write(s);
        }

    }
}
